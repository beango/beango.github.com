/* 正则检查表单案例 */
function checkform(){
    $element = document.getElementById("telephone");
    $m = $element.value.match(/^((\(\d{2,3}\))|(\d{3}\-))?(\(0\d{2,3}\)|0\d{2,3}-)?[1-9]\d{6,7}(\-\d{1,4})?$/);
    if($m == null){
        alert("电话号码不正确");return false;
    }
    else{
        alert("恭喜！电话号码正确");return false;
    }
}
